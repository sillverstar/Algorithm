# 방향 벡터: 시뮬레이션 및 완점 탐색

# 동(0, 1) 북(-1, 0) 서(0, -1) 남(1, 0)
dx = [0, -1, 0, 1]
dy = [1, 0, -1, 0]

x, y = 2, 2

for i in range(4):
    nx = x + dx[i]
    ny = y + dy[i]
    print(nx, ny)
